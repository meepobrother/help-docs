AnimationPlayer


export interface AnimationPlayer {
    onDone(fn: () => void): void;
    onStart(fn: () => void): void;
    onDestroy(fn: () => void): void;
    init(): void;
    hasStarted(): boolean;
    play(): void;
    pause(): void;
    restart(): void;
    finish(): void;
    destroy(): void;
    reset(): void;
    setPosition(p: any): void;
    getPosition(): number;
    parentPlayer: AnimationPlayer | null;
    readonly totalTime: number;
    beforeDestroy?: () => any;
}
